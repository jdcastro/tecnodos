460        Returns:

470        Create a new recommendation
471        Returns:
472            JSON: Details of the created recommendation
473        """
474        data = request.get_json()
475        required_fields = ["lot_id", "date", "recommendation"]

482        """
483        Update an existing recommendation
484        Args:

486        Returns:
487            JSON: Details of the updated recommendation
488        """
489        data = request.get_json()

499        Args:

501        Returns:
502            JSON: Confirmation message

509    # Helper Methods
510    def _get_recommendation_list(self):
511        """Retrieve a list of all recommendations"""
512        claims = get_jwt()
513        user_role = claims.get("rol")
514        if user_role == RoleEnum.ADMINISTRATOR.value:
515            recommendations = Recommendation.query.all()
516        elif user_role == RoleEnum.RESELLER.value:
517            reseller_package = ResellerPackage.query.filter_by(
518                reseller_id=claims.get("org_id")
519            ).first()
520            if not reseller_package:
521                raise NotFound("Reseller package not found.")
522            recommendations = []
523            for organization in reseller_package.organizations:
524                for lot in organization.lots:
525                    recommendations.extend(lot.recommendations)
526        else:
527            raise Forbidden(
528                "Only administrators and resellers can list recommendations"
529            )
530        response_data = [self._serialize_recommendation(r) for r in recommendations]
531        json_data = json.dumps(response_data, ensure_ascii=False, indent=4)
532        return Response(json_data, status=200, mimetype="application/json")
